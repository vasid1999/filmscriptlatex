% To use Courier font
\renewcommand{\familydefault}{\ttdefault}

% Script title. Temporary; will have to be standardised later
\newcommand{\makescripttitle}{
    \pagenumbering{gobble}
    \newgeometry{
    a4paper,
    left=1in,
    top=1in,
    right=1in,
    bottom=1in
    }
    \maketitle
    \newgeometry{
    a4paper,
    left=1.5in,
    top=1in,
    right=1in,
    bottom=1in
    }
    \pagenumbering{arabic}
    \newpage
}

% Scene heading. Bold, large, all CAPS.
\newcommand{\scene}[1]{\label{#1}\noindent\large \textbf{\MakeTextUppercase{#1}}\\\normalsize}

% Action para. Normal.
\newcommand{\action}[1]{
    \begin{adjustwidth}{0in}{1in}
    \noindent #1 \\
    \end{adjustwidth}
}

% Character name. Capitalised. Need to create a global character index.
\newcommand{\charname}[1]{\noindent\MakeTextUppercase{#1}}

% Dialogue Title. 3.7 inches from left edge.
\newcommand{\dialoguetitle}[1]{
    \begin{adjustwidth}{2.2in}{0in}#1\end{adjustwidth}
}

% Parenthetical (for emotion). Put into brackets.
\newcommand{\parenthetical}[1]{
    \begin{adjustwidth}{1.7in}{0in}
    (#1)
    \end{adjustwidth}}

% Dialogue. 2.5 inches from left edge and 3.7 inches from right edge.
\newcommand{\dialogue}[1]{
    \begin{adjustwidth}{1in}{2.2in}
    #1\\
    \end{adjustwidth}
}
